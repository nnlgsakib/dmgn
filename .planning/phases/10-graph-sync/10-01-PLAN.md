```go
func (dm *DeltaSyncManager) collectMissingEdges(missing map[string]uint64) [][]byte {
    var edges [][]byte
    for peerID, afterSeq := range missing {
        edgeKeys, err := dm.vvStore.GetEdgesAfter(peerID, afterSeq)
        if err != nil {
            continue
        }
        for _, edgeKey := range edgeKeys {
            parts := strings.SplitN(edgeKey, ":", 2)
            if len(parts) != 2 {
                continue
            }
            // Reconstruct edge from storage
            edgeData, err := dm.store.GetEdgeProto(parts[0], parts[1])
            if err != nil {
                continue
            }
            edges = append(edges, edgeData)
        }
    }
    return edges
}
```

## Task 6: Wire edge broadcast in daemon and MCP handler

**Files:** `internal/daemon/daemon.go`, `pkg/mcp/server.go`

**In daemon.go** — create `broadcastEdge` function (analogous to `broadcastMemory`):

```go
broadcastEdge := func(fromID, toID string, weight float32, edgeType string) {
    if d.gossipMgr == nil {
        return
    }
    edge := &dmgnpb.Edge{
        FromId:        fromID,
        ToId:          toID,
        Weight:        weight,
        EdgeType:      edgeType,
        Timestamp:     time.Now().UnixNano(),
        CreatorPeerId: localPeerID,
    }
    data, err := proto.Marshal(edge)
    if err != nil {
        d.logger.Error("failed to marshal edge for gossip", "from", fromID, "to", toID, "err", err)
        return
    }
    edgeSeq := edgeVV.Increment("edge:" + localPeerID)
    vvStore.SaveEdgeSequence(localPeerID, edgeSeq, fromID+":"+toID)
    vvStore.Save(localPeerID, edgeVV)
    if err := d.gossipMgr.PublishEdge(d.nodeCtx, data, edgeSeq); err != nil {
        d.logger.Error("failed to broadcast edge", "from", fromID, "to", toID, "err", err)
    } else {
        d.logger.Info("edge broadcast to network", "from", fromID, "to", toID, "type", edgeType)
    }
}
```

Wire `onEdgeReceived` callback into gossip manager:
```go
onEdgeReceived := func(msg *dmgnpb.GossipMessage) {
    pb := &dmgnpb.Edge{}
    if err := proto.Unmarshal(msg.Edge, pb); err != nil {
        return
    }
    // Validate both memories exist
    if _, err := d.store.GetMemory(pb.FromId); err != nil {
        // Buffer for retry (pending edges)
        return
    }
    if _, err := d.store.GetMemory(pb.ToId); err != nil {
        return
    }
    if err := d.store.SaveEdgeWithMeta(pb); err != nil {
        return
    }
    graph := d.store.GetGraph()
    graph.AddEdge(pb.FromId, pb.ToId, pb.Weight, pb.EdgeType)
    d.logger.Info("edge received from peer", "from", pb.FromId, "to", pb.ToId)
}
```
