```go
// parseRelayAddrs converts multiaddr strings to peer.AddrInfo for static relay configuration.
func parseRelayAddrs(addrs []string) ([]peer.AddrInfo, error) {
	var infos []peer.AddrInfo
	for _, addr := range addrs {
		ma, err := multiaddr.NewMultiaddr(addr)
		if err != nil {
			continue
		}
		pi, err := peer.AddrInfoFromP2pAddr(ma)
		if err != nil {
			continue
		}
		infos = append(infos, *pi)
	}
	if len(infos) == 0 {
		return nil, fmt.Errorf("no valid relay addresses")
	}
	return infos, nil
}
```

Note: The `peer` import (`github.com/libp2p/go-libp2p/core/peer`) is already used in the file. The `relay` import from `circuitv2` is not used directly in this minimal approach since `libp2p.EnableRelayService()` uses defaults. Only add the `relay` import if customizing resources.
