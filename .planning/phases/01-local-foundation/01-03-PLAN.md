     func writeLenPrefixedProto(w io.Writer, msg proto.Message) error {
         data, err := proto.Marshal(msg)
         if err != nil { return err }
         lenBuf := make([]byte, 4)
         binary.BigEndian.PutUint32(lenBuf, uint32(len(data)))
         if _, err := w.Write(lenBuf); err != nil { return err }
         _, err = w.Write(data)
         return err
     }

     func readLenPrefixedProto(r io.Reader, msg proto.Message) error {
         lenBuf := make([]byte, 4)
         if _, err := io.ReadFull(r, lenBuf); err != nil { return err }
         size := binary.BigEndian.Uint32(lenBuf)
         if size > 10*1024*1024 { return fmt.Errorf("message too large: %d", size) }
         data := make([]byte, size)
         if _, err := io.ReadFull(r, data); err != nil { return err }
         return proto.Unmarshal(data, msg)
     }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GossipSub message → proto.Unmarshal | Any peer on the topic can broadcast gossip messages |
| sync stream → proto.Unmarshal | Connected peer sends sync request/response |
| query stream → proto.Unmarshal | Connected peer sends query request |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-1-07 | D (DoS) | gossip.Start | mitigate | libp2p-pubsub has built-in message size limits (1MB default). Proto Unmarshal is bounded by message size. No change from JSON baseline. |
| T-1-08 | D (DoS) | delta.handleStream | mitigate | `readLenPrefixedProto` enforces 10MB max size before allocating buffer. Prevents memory exhaustion from malicious length prefix. |
| T-1-09 | T (Tampering) | delta.processReceivedMemories | accept | Received memories are validated by Memory.VerifyIntegrity() in the onReceive callback. Proto deserialization does not bypass integrity checks. |
| T-1-10 | I (Info Disclosure) | query.RegisterQueryHandler | accept | Query results contain only encrypted payloads and metadata (no plaintext). Same privacy boundary as JSON path. |
| T-1-11 | D (DoS) | query.readLenPrefixedProto | mitigate | 10MB max size limit in readLenPrefixedProto. queryStreamTimeout (5s) unchanged. |
</threat_model>
