```go
package query

const QueryProtocol = protocol.ID("/dmgn/memory/query/1.0.0")

// QueryProtocolRequest is the wire format for cross-peer queries.

}

// RegisterQueryHandler registers the /dmgn/memory/query/1.0.0 stream handler.
func RegisterQueryHandler(h host.Host, engine *QueryEngine, localPeerID string)

// QueryPeer sends a query to a specific peer and returns results.
func QueryPeer(ctx context.Context, h host.Host, peerID peer.ID, req QueryProtocolRequest) (*QueryProtocolResponse, error)
```

Implementation details:
- `RegisterQueryHandler`: on incoming stream, read request frame, call `engine.SearchLocal()`, set SourcePeer to local peer ID, send response frame
- `QueryPeer`: open stream, write request, read response, return results
- Use existing `writeFrame`/`readFrame` from `pkg/network/protocols.go` ŌĆö import as helper or duplicate pattern

### Task 2: Remote Query Orchestrator

**File:** `pkg/query/remote.go`

```go
// RemoteQueryOrchestrator fans out queries to connected peers.
type RemoteQueryOrchestrator struct {
    host         host.Host
    engine       *QueryEngine
    localPeerID  string
    queryTimeout time.Duration
}

// NewRemoteQueryOrchestrator creates a new orchestrator.
func NewRemoteQueryOrchestrator(h host.Host, engine *QueryEngine,
    localPeerID string, queryTimeout time.Duration) *RemoteQueryOrchestrator

// SearchAll performs a local search and fans out to all connected peers.
// Results are merged with source diversity and deduplication.
func (ro *RemoteQueryOrchestrator) SearchAll(ctx context.Context, req QueryRequest) ([]QueryResult, error)
```

Fan-out implementation:
1. Start local search in goroutine
2. Get connected peers from host
3. Send query to each peer in parallel goroutines (per-peer timeout from config)
4. Collect results into channel
5. Merge all results:
   - Deduplicate by memory_id (keep highest score)
   - Group by source_peer
   - Interleave: round-robin pick from each peer's results
   - Return top Limit

### Task 5: Wire into Node Startup

**File:** `internal/cli/start.go` (modify existing)

Add after existing shard setup:

```go
// 1. Derive vector index encryption key
indexKey, err := id.DeriveKey("vector-index", 32)
indexCrypto, err := crypto.NewEngine(indexKey)

// 2. Create and load vector index
vecIndex := vectorindex.NewVectorIndex(
    cfg.VectorIndexPath(),
    indexCrypto.Encrypt,
    indexCrypto.Decrypt,
)
vecIndex.Load()  // load existing or start empty

// 3. Create query engine
queryEngine := query.NewQueryEngine(vecIndex, store, decryptFn, cfg.HybridScoreAlpha)

// 4. Create remote query orchestrator
remoteOrch := query.NewRemoteQueryOrchestrator(h, queryEngine, h.ID().String(), cfg.QueryTimeoutDuration())

// 5. Register query protocol handler
query.RegisterQueryHandler(h.LibP2PHost(), queryEngine, h.ID().String())

// 6. Set up version vector
vvStore := sync.NewVClockStore(store.DB())
vv, _ := vvStore.Load(h.ID().String())

// 7. Memory receive callback (shared by gossip + delta sync)
onMemoryReceived := func(mem *memory.Memory) {
    store.SaveMemory(mem)
    if len(mem.Embedding) > 0 {
        vecIndex.Add(mem.ID, mem.Embedding)
    }
}

// 8. Start gossip manager
gossipMgr, err := sync.NewGossipManager(ctx, h.LibP2PHost(), cfg.GossipTopic,
    h.ID().String(), func(msg *sync.GossipMessage) {
        mem, _ := memory.FromJSON(msg.Memory)
        onMemoryReceived(mem)
        vv.Merge(...)  // update version vector
    })
gossipMgr.Start(ctx)

// 9. Start delta sync manager
deltaMgr := sync.NewDeltaSyncManager(h.LibP2PHost(), vv, vvStore, store,
    h.ID().String(), cfg.SyncIntervalDuration(), onMemoryReceived)
deltaMgr.RegisterHandler()
deltaMgr.Start(ctx)

// 10. Wire query engine and gossip into API server
server.SetQueryEngine(queryEngine, remoteOrch)
server.SetGossipManager(gossipMgr)

// 11. Ensure index is saved on shutdown
defer vecIndex.Save()
```
